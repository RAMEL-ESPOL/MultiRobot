48ArucoMarkerPublisher::ArucoMarkerPublisher(rclcpp::NodeOptions options)
49: cam_info_received(false),
50 useCamInfo_(true),
51 Node("aruco_marker_publisher", options),
52 tf_buffer_(std::make_shared<tf2_ros::Buffer>(get_clock()))
53{
54 auto timer_interface = std::make_shared<tf2_ros::CreateTimerROS>(
55 get_node_base_interface(),
56 get_node_timers_interface());
57 tf_buffer_->setCreateTimerInterface(timer_interface);
58 tfListener_ = std::make_shared<tf2_ros::TransformListener>(*tf_buffer_);
59
60 image_sub_ =
61 image_transport::create_subscription(
62 this, "/image",
63 std::bind(&ArucoMarkerPublisher::image_callback, this, std::placeholders::_1), "raw");
64
65 get_parameter_or("use_camera_info", useCamInfo_, true);
66
67 cam_info_sub = create_subscription<sensor_msgs::msg::CameraInfo>(
68 "/camera_info",
69 rclcpp::SensorDataQoS(),
70 std::bind(&ArucoMarkerPublisher::cam_info_callback, this, std::placeholders::_1));
71 if (useCamInfo_) {
72 get_parameter_or("marker_size", marker_size_, 0.1);
73 get_parameter_or("image_is_rectified", useRectifiedImages_, false);
74 get_parameter_or("reference_frame", reference_frame_, string{""});
75 get_parameter_or("camera_frame", camera_frame_, string{"camera"});
76 get_parameter_or("rotate_marker_axis", rotate_marker_axis_, false);
77 if (reference_frame_.empty()) {
78 reference_frame_ = camera_frame_;
79 }
80 } else {
81 camParam_ = aruco::CameraParameters();
82 }
83 image_pub_ = image_transport::create_publisher(this, "result");
84 debug_pub_ = image_transport::create_publisher(this, "debug");
85 marker_pub_ = create_publisher<aruco_msgs::msg::MarkerArray>("markers", 100);
86 marker_list_pub_ = create_publisher<std_msgs::msg::UInt32MultiArray>("markers_list", 10);
87 marker_msg_ = aruco_msgs::msg::MarkerArray::Ptr(new aruco_msgs::msg::MarkerArray());
88 marker_msg_->header.frame_id = reference_frame_;
89}

91bool ArucoMarkerPublisher::getTransform(
92 const std::string & refFrame,
93 const std::string & childFrame,
94 geometry_msgs::msg::TransformStamped & transform)
95{
96 std::string errMsg;
97 if (!tf_buffer_->canTransform(
98 refFrame,
99 childFrame,
100 tf2::TimePointZero,
101 tf2::durationFromSec(1.0),
102 &errMsg))
103 {
104 RCLCPP_ERROR_STREAM(get_logger(), "Unable to get pose from TF: " << errMsg);
105 return false;
106 } else {
107 try {
108 transform = tf_buffer_->lookupTransform(
109 refFrame, childFrame,
110 tf2::TimePointZero);
111 } catch (const tf2::TransformException & e) {
112 RCLCPP_ERROR_STREAM(
113 get_logger(), "Error in lookupTransform of " << childFrame << " in " << refFrame);
114 return false;
115 }
116 }
117 return true;
118}
