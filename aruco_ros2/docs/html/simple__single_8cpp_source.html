137bool ArucoSimple::getTransform(
138 const std::string & refFrame,
139 const std::string & childFrame,
140 geometry_msgs::msg::TransformStamped & transform)
141{
142 std::string errMsg;
143 if (!tf_buffer_->canTransform(
144 refFrame,
145 childFrame,
146 rclcpp::Time(0),
147 rclcpp::Duration(0.5),
148 &errMsg))
149 {
150 RCLCPP_ERROR_STREAM(get_logger(), "Unable to get pose from TF: " << errMsg << " YY");
151 return false;
152 } else {
153 try {
154 transform = tf_buffer_->lookupTransform(
155 refFrame, childFrame,
156 rclcpp::Time(0),
157 rclcpp::Duration(0.5));
158 } catch (const tf2::TransformException & e) {
159 RCLCPP_ERROR_STREAM(
160 get_logger(), "Error in lookupTransform of " << childFrame << " in " << refFrame);
161 return false;
162 }
163 }
164 return true;
165}

167void ArucoSimple::image_callback(const sensor_msgs::msg::Image::ConstPtr & msg)
168{
169 if (cam_info_received) {
170 rclcpp::Time curr_stamp(now());
171 cv_bridge::CvImagePtr cv_ptr;
172 try {
173 cv_ptr = cv_bridge::toCvCopy(msg, sensor_msgs::image_encodings::RGB8);
174 inImage = cv_ptr->image;
175 // detection results will go into "markers"
176 markers.clear();
177 // Ok, let's detect
178 mDetector.detect(inImage, markers, camParam, marker_size, false);
179 // for each marker, draw info and its boundaries in the image
180 for (size_t i = 0; i < markers.size(); ++i) {
181 // only publishing the selected marker
182 if (markers[i].id == marker_id) {
183 tf2::Transform transform = aruco_ros::arucoMarker2Tf(markers[i], rotate_marker_axis_);
184 geometry_msgs::msg::TransformStamped cameraToReference;
185 tf2::Transform cameraToReference_tf = tf2::Transform::getIdentity();
186 // <false,true>=> tf to msg, <true,false>=> msg to tf
187 tf2::impl::Converter<false, true>::convert(
188 cameraToReference_tf,
189 cameraToReference.transform);
190 if (reference_frame != camera_frame) {
191 if (!getTransform(reference_frame, camera_frame, cameraToReference)) {
192 return;
193 }
194 }
195 tf2::impl::Converter<true, false>::convert(
196 cameraToReference.transform,
197 cameraToReference_tf);
198 transform =
199 static_cast<tf2::Transform>(cameraToReference_tf) *
200 static_cast<tf2::Transform>(rightToLeft) * transform;
201 geometry_msgs::msg::TransformStamped stampedTransform;
202 stampedTransform.header.frame_id = reference_frame;
203 stampedTransform.header.stamp = curr_stamp;
204 stampedTransform.child_frame_id = marker_frame;
205 tf2::impl::Converter<false, true>::convert(transform, stampedTransform.transform);
206 br->sendTransform(stampedTransform);
207 geometry_msgs::msg::PoseStamped poseMsg;
208 tf2::toMsg(transform, poseMsg.pose);
209 poseMsg.header.frame_id = reference_frame;
210 poseMsg.header.stamp = curr_stamp;
211 pose_pub->publish(poseMsg);
212 transform_pub->publish(stampedTransform);
213 geometry_msgs::msg::Vector3Stamped positionMsg;
214 positionMsg.header = stampedTransform.header;
215 positionMsg.vector = stampedTransform.transform.translation;
216 position_pub->publish(positionMsg);
217 geometry_msgs::msg::PointStamped pixelMsg;
218 pixelMsg.header = stampedTransform.header;
219 pixelMsg.point.x = markers[i].getCenter().x;
220 pixelMsg.point.y = markers[i].getCenter().y;
221 pixelMsg.point.z = 0;
222 pixel_pub->publish(pixelMsg);
223 // Publish rviz marker representing the ArUco marker patch
224 visualization_msgs::msg::Marker visMarker;
225 visMarker.header = stampedTransform.header;
226 visMarker.id = 1;
227 visMarker.type = visualization_msgs::msg::Marker::CUBE;
228 visMarker.action = visualization_msgs::msg::Marker::ADD;
229 visMarker.pose = poseMsg.pose;
230 visMarker.scale.x = marker_size;
231 visMarker.scale.y = 0.001;
232 visMarker.scale.z = marker_size;
233 visMarker.color.r = 1.0;
234 visMarker.color.g = 0;
235 visMarker.color.b = 0;
236 visMarker.color.a = 1.0;
237 visMarker.lifetime = rclcpp::Duration(3.0);
238 marker_pub->publish(visMarker);
239 }
240 // but drawing all the detected markers
241 markers[i].draw(inImage, cv::Scalar(0, 0, 255), 2);
242 }
243 // draw a 3d cube in each marker if there is 3d info
244 if (camParam.isValid() && marker_size > 0) {
245 for (size_t i = 0; i < markers.size(); ++i) {
246 aruco::CvDrawingUtils::draw3dAxis(inImage, markers[i], camParam);
247 }
248 }
249 if (image_pub.getNumSubscribers() > 0) {
250 // show input with augmented information
251 cv_bridge::CvImage out_msg;
252 out_msg.header.stamp = curr_stamp;
253 out_msg.encoding = sensor_msgs::image_encodings::RGB8;
254 out_msg.image = inImage;
255 image_pub.publish(out_msg.toImageMsg());
256 }
257 // cv::imshow("display check", inImage);
258 // cv::waitKey(1);
259 if (debug_pub.getNumSubscribers() > 0) {
260 // show also the internal image resulting from the threshold operation
261 cv_bridge::CvImage debug_msg;
262 debug_msg.header.stamp = curr_stamp;
263 debug_msg.encoding = sensor_msgs::image_encodings::MONO8;
264 debug_msg.image = mDetector.getThresholdedImage();
265 debug_pub.publish(debug_msg.toImageMsg());
266 }
267 } catch (cv_bridge::Exception & e) {
268 std::cout << "try failed" << std::endl;
269 RCLCPP_ERROR(get_logger(), "cv_bridge exception: %s", e.what());
270 return;
271 }
272 }

275void ArucoSimple::cam_info_callback(sensor_msgs::msg::CameraInfo::ConstPtr msg)
276{
277 camParam = aruco_ros::rosCameraInfo2ArucoCamParams(*msg, useRectifiedImages);

279 // see the sensor_msgs/CamaraInfo documentation for details
